```typescript
import { Interface } from "reliant-type";

const UserRegistrationSchema = Interface({
  // Basic information
  email: "email",
  password: "string(8,128)",
  confirmPassword: "string(8,128)",

    firstName: "string(1,50)",
    lastName: "string(1,50)",
    displayName: "string(3,50)?",
    dateOfBirth: "date?",
    gender: "male|female|other|prefer-not-to-say"?,
    avatar: "url?",
    bio: "string(,500)?"
  },
  
  // Contact information
  contact: {
    phone: "phone?",
    address: {
      street: "string(5,100)",
      city: "string(2,50)",
      state: "string(2,50)",
      zipCode: "string(/^\\d{5}(-\\d{4})?$/)",
      country: "string(/^[A-Z]{2}$/)"
    }?
  }?,
  
  // Preferences
  preferences: {
    theme: "light|dark|auto",
    language: "string(/^[a-z]{2}(-[A-Z]{2})?$/)",
    timezone: "string?",
    notifications: {
      email: "boolean",
      push: "boolean",
      sms: "boolean",
      marketing: "boolean"
    }
  },
  
  // Legal agreements
  agreements: {
    termsOfService: "=true",
    privacyPolicy: "=true",
    marketingConsent: "boolean",
    dataProcessing: "=true"
  },
  
  // Runtime configuration for conditional validation
  config: "any?",
  
  // Conditional fields
  betaFeatures: "when config.beta.$exists() *? string[] : =[]",
  referralCode: "when config.referral.$exists() *? string : =null"
});

// Usage example
```
